sql02_create_tables.sql
CREATE TABLE Usuario (
    id_usuario INT PRIMARY KEY,
    nombre VARCHAR(100),
    correo VARCHAR(100),
    fecha_registro DATE
);

CREATE TABLE Instructor (
    id_instructor INT PRIMARY KEY,
    nombre VARCHAR(100),
    especialidad VARCHAR(100)
);

CREATE TABLE Categoria (
    id_categoria INT PRIMARY KEY,
    nombre VARCHAR(100)
);

CREATE TABLE Curso (
    id_curso INT PRIMARY KEY,
    titulo VARCHAR(100),
    id_instructor INT,
    FOREIGN KEY (id_instructor) REFERENCES Instructor(id_instructor)
);

CREATE TABLE Curso_Categoria (
    id_curso INT,
    id_categoria INT,
    PRIMARY KEY (id_curso, id_categoria),
    FOREIGN KEY (id_curso) REFERENCES Curso(id_curso),
    FOREIGN KEY (id_categoria) REFERENCES Categoria(id_categoria)
);

CREATE TABLE Leccion (
    id_leccion INT PRIMARY KEY,
    titulo VARCHAR(100),
    id_curso INT,
    FOREIGN KEY (id_curso) REFERENCES Curso(id_curso)
);

CREATE TABLE Visualizacion (
    id_usuario INT,
    id_leccion INT,
    fecha DATE,
    PRIMARY KEY (id_usuario, id_leccion),
    FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario),
    FOREIGN KEY (id_leccion) REFERENCES Leccion(id_leccion)
);

CREATE TABLE Certificado (
    id_certificado INT PRIMARY KEY,
    id_usuario INT,
    id_curso INT,
    fecha DATE,
    FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario),
    FOREIGN KEY (id_curso) REFERENCES Curso(id_curso)
);
